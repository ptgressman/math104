\section*{Examples}

\begin{example}
(Spring 2016 Final Exam) Find the volume of the solid generated by revolving the region in the first quadrant bounded by $y = 1-x^2$, $y = 0$, and $x = 0$ about the line $x=2$.
\begin{itemize}
\item Using $x$ as the slicing variable, slices are \wordChoice{\choice[correct]{parallel}\choice{perpendicular}} to the axis $x=2$. This alignment corresponds to the \wordChoice{\choice{disk or washer}\choice[correct]{shell}} method.
(Expand to continue.)
\end{itemize}
\begin{expandable}
\begin{itemize}
\item The height of a shell is 
\begin{multipleChoice}
\choice{$2$}
\choice{$2-x$}
\choice{$0$}
\choice{$x-2$}
\choice[correct]{$1-x^2$}
\choice{$1 - x^2 - 2$}
\end{multipleChoice}
and the radius is
\begin{multipleChoice}
\choice{$2$}
\choice[correct]{$2-x$}
\choice{$0$}
\choice{$x-2$}
\choice{$1-x^2$}
\choice{$1 - x^2 - 2$}
\end{multipleChoice}
\end{itemize}
Therefore
\[ V = \int_{\answer{0}}^{\answer{1}} \answer{2 \pi (2-x)(1-x^2)} dx = \answer{\frac{13 \pi}{6}}. \]
\end{expandable}
\end{example}

\begin{example}
(Fall 2012 Final Exam) The region in the $xy$-plane bounded by $y = (x-1)^{1/4}$ and the $x$-axis for $1 \leq x \leq 2$ is rotated about the $x$-axis. What is the volume of the resulting solid of revolution?
\begin{itemize}
\item Using $x$ as the slicing variable, slices are \wordChoice{\choice{parallel}\choice[correct]{perpendicular}} to the axis $x=2$. This alignment corresponds to the \wordChoice{\choice[correct]{disk or washer}\choice{shell}} method.
(Expand to continue.)
\end{itemize}
\begin{expandable}
\begin{itemize}
\item The outer radius of a disk is 
\begin{multipleChoice}
\choice{$2$}
\choice{$y^4+1$}
\choice{$0$}
\choice[correct]{$(x-1)^{1/4}$}
\end{multipleChoice}
and the inner radius is
\begin{multipleChoice}
\choice{$2$}
\choice{$y^4+1$}
\choice[correct]{$0$}
\choice{$(x-1)^{1/4}$}
\end{multipleChoice}
\end{itemize}
Therefore
\[ V = \int_{\answer{1}}^{\answer{2}} \answer{\pi (x-1)^{1/2}} dx = \answer{\frac{2 \pi}{3}}. \]
\end{expandable}
\end{example}

\begin{example}
Find the volume of the solid generated by revolving the region in the first quadrant bounded on the left by $y = \sqrt{x}$,  on the right by $y = x-2$, and below by $y = 0$ about the axis $y = 0$.
\begin{itemize}
\item To carry out the calculation as efficiently as possible, the \wordChoice{\choice{$x$}\choice[correct]{$y$}}-variable should be used as the variable for slicing.
(Expand to continue.)
\end{itemize}
\begin{itemize}
\item Using $y$ as the slicing variable, slices are \wordChoice{\choice[correct]{parallel}\choice{perpendicular}} to the axis $y=0$. This alignment corresponds to the \wordChoice{\choice{disk or washer}\choice[correct]{shell}} method.
(Expand to continue.)
\end{itemize}
\begin{expandable}
\begin{itemize}
\item The height of a shell is 
\begin{multipleChoice}
\choice{$y + 2$}
\choice{$y^2$}
\choice{$x-2$}
\choice{$\sqrt{x}$}
\choice{$y^2-y-2$}
\choice[correct]{$y+2-y^2$}
\end{multipleChoice}
and the radius is
\begin{multipleChoice}
\choice[correct]{$y$}
\choice{$y^2$}
\choice{$x-2$}
\choice{$\sqrt{x}$}
\choice{$y^2-y-2$}
\choice{$y+2-y^2$}
\end{multipleChoice}
\end{itemize}
Therefore
\[ V = \int_{\answer{0}}^{\answer{2}} \answer{2 \pi y(y+2-y^2)} dy = \answer{\frac{16 \pi}{3}}. \]
\end{expandable}
\end{example}

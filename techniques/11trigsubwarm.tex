\[ \int \sqrt{9 - x^2} dx. \]
\begin{itemize}
\item The structure of this integrand is adapted to a trigonometric substitution of \wordChoice{\choice{secant}\choice[correct]{sine}\choice{tangent}} type, so we let $x = 3 \answer{\sin \theta}$ (to type out the word \code{theta}).
\end{itemize}
\end{example}
